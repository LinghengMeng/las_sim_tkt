62 void read_EE(uint8_t word, uint8_t *buf, uint8_t offset)
63 {
64  noInterrupts();
65  FTFL_FCCOB0 = 0x41; // Selects the READONCE command
66  FTFL_FCCOB1 = word; // read the given word of read once area
67 
68  // launch command and wait until complete
69  FTFL_FSTAT = FTFL_FSTAT_CCIF;

73  *(buf + offset + 1) = FTFL_FCCOB5;
