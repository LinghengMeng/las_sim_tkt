488  masktop = 0 - ( min(0, py-cornerdist ));
489  maskright = 0 - ( min(0, ((msize-1)-px)-cornerdist ));
490  maskbot = 0 - ( min(0, ((msize-1)-py)-cornerdist ));

492

496  windex / wsize >= masktop && (windex / wsize) < (wsize-maskbot) ) {
