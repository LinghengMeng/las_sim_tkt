305  boolean containsPoint(PVector[] verts, float px, float py) {
306  int num = verts.length;
307  int i, j = num - 1;
308  boolean oddNodes = false;
309  for (i = 0; i < num; i++) {
310  PVector vi = verts[i];
311  PVector vj = verts[j];
312  if (vi.y < py && vj.y >= py || vj.y < py && vi.y >= py) {
313  if (vi.x + (py - vi.y) / (vj.y - vi.y) * (vj.x - vi.x) < px) {
314  oddNodes = !oddNodes;
315  }
316  }
317  j = i;
318  }
319  return oddNodes;
320  }
