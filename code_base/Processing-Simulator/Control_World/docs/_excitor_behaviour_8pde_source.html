206  public void setActuatorDistanceIntensities() {
207 
208  if (excitorBehaviour.excitorsInfluenceActive) {
209  for (int i=0; i<excitorBehaviour.actuatorLocations.size(); i++) { //iterate through all actuators (140 for Futurium)
210  float distance = location.dist(actuatorLocations.get(i)); //calculate distance between thisExcitor and actuators
211  float currentIntensity = excitorBehaviour.actuatorIntensities.get(i); //intensities per actuator are cumulative, add newIntensity on top of existing value (caused by other excitors)
212  float thisIntensity;
213  float newIntensity;
214  if (distance<threshold) { //only calculate intensity when distance is smaller than threshold (within 'reach')
215  thisIntensity = 1 - distance/threshold; //calculate intensity of thisExcitor
216  newIntensity = (currentIntensity + thisIntensity) * presence; //resulting intensity is (currentIntensity + thisIntensity) * presence
217  if (newIntensity>maxBrightness) newIntensity = maxBrightness; //intensity cannot exceed 1
218  excitorBehaviour.actuatorIntensities.set(i, newIntensity); //set value in actuatorIntensities ArrayList
219  } else {
220  excitorBehaviour.actuatorIntensities.set(i, currentIntensity); //when thisExcitor is not within threshold then use currentIntensity
221  }
222  }
223  }
224  if (!excitorBehaviour.excitorsInfluenceActive) { //if excitors are not active then set all intensities to 0
225  for (int i=0; i<excitorBehaviour.actuatorLocations.size(); i++) {
226  excitorBehaviour.actuatorIntensities.set(i, 0.);
227  }
228  }
229  }
